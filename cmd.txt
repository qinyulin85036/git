git init   //初始化本地git仓库
git add <file>  //添加文件
git status //查看状态
git commit -m ''   //提交注释

git pull --rebase origin master           //从远程仓库拉取数据
git clone          // 从远程仓库拷贝数据
git remote add origin <url>  //在本地增加远程仓库的地址
git remote remove origin     //删除本地仓库地址
git push -u origin master    //把本地仓库的推送到远程仓库的主干上去

.gitignore    //忽略文件集合文件名(文件夹的话前面加/)   
touch         //新建文件
